# 芯师爷官网 — 数据模型设计

## 一、数据建模原则

### 1.1 核心设计思想

- **Company 为中心**：企业是平台的核心实体，所有业务模块围绕企业展开
- **增量扩展**：基础模型优先，增值模型逐步追加
- **JSON 字段灵活**：对于频繁变化的元数据，使用 JSON 类型而非新增表
- **软删除**：所有业务数据支持逻辑删除（`deletedAt` 字段）

### 1.2 命名规范

| 规范 | 示例 |
|------|------|
| 模型名：大驼峰、单数 | `Company`、`SoftArticle` |
| 字段名：小驼峰 | `companyId`、`publishedAt` |
| 枚举名：大驼峰 | `ArticleStatus`、`MetricType` |
| 枚举值：全大写下划线 | `PENDING_REVIEW`、`INDUSTRY_TREND` |
| 关联字段：模型名 + Id | `companyId`、`awardId` |
| 时间字段：动词 + At | `createdAt`、`updatedAt`、`publishedAt` |

## 二、完整数据模型

### 2.1 核心企业模型

```prisma
// 企业系统（核心）
model Company {
  id          String   @id @default(cuid())
  name        String   @unique
  slug        String   @unique
  logo        String?
  description String?  // 企业介绍（支持 Markdown）
  website     String?
  industry    String?  // 行业分类（IC设计/制造/封测/EDA/设备/材料）
  scale       String?  // 规模（<50/50-200/200-1000/1000+）
  city        String?  // 所在城市
  socialLinks Json?    // { "wechat": "...", "linkedin": "...", "twitter": "..." }
  status      Status   @default(PENDING)
  deletedAt   DateTime?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // 关联
  products     Product[]
  news         CompanyNews[]
  events       Event[]
  jobs         Job[]
  awards       Award[]
  ads          Ad[]
  reports      Report[]
  mediaKit     MediaKit?
  brandAssets  BrandAsset[]
  caseStudies  CaseStudy[]
  dataMetrics  DataMetric[]
  softArticles SoftArticle[]
  adCampaigns  AdCampaign[]
}

enum Status {
  PENDING    // 待审核
  APPROVED   // 已通过
  REJECTED   // 已拒绝
  SUSPENDED  // 已暂停
}
```

### 2.2 产品系统

```prisma
model Product {
  id          String   @id @default(cuid())
  companyId   String
  company     Company  @relation(fields: [companyId], references: [id])
  name        String
  category    String?  // 产品分类（MCU/传感器/电源/存储等）
  description String?  // 产品描述
  parameters  Json?    // { "process": "5nm", "package": "BGA", ... }
  images      Json?    // 产品图片列表
  datasheetUrl String? // 数据手册下载链接
  status      Status   @default(APPROVED)
  sort        Int      @default(0)  // 排序权重
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  @@index([companyId])
  @@index([category])
}
```

### 2.3 内容系统

```prisma
model Article {
  id            String   @id @default(cuid())
  title         String
  slug          String   @unique
  summary       String?  // 摘要
  content       String   // Markdown 正文
  coverImage    String?
  category      String?  // 行业动态/技术解读/市场分析/企业报道
  tags          String[]
  author        String?
  source        String?  // 来源（芯师爷原创/企业投稿）
  companyId     String?
  company       Company? @relation(fields: [companyId], references: [id])
  isFeatured    Boolean  @default(false)  // 是否推荐
  viewCount     Int      @default(0)
  status        ArticleStatus @default(DRAFT)
  publishedAt   DateTime?
  deletedAt     DateTime?
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  @@index([status, publishedAt])
  @@index([category])
  @@index([slug])
}

enum ArticleStatus {
  DRAFT
  PENDING_REVIEW
  APPROVED
  REJECTED
  PUBLISHED
  ARCHIVED
}
```

### 2.4 软文/媒体发布系统

```prisma
model SoftArticle {
  id           String        @id @default(cuid())
  companyId    String
  company      Company       @relation(fields: [companyId], references: [id])
  title        String
  content      String
  summary      String?       // SEO 摘要
  status       ArticleStatus @default(DRAFT)
  channels     Json?         // 发布渠道列表 ["公众号", "头条号", "知乎", "行业门户", ...]
  seoKeywords  String?
  publishedAt  DateTime?
  createdAt    DateTime      @default(now())
  updatedAt    DateTime      @updatedAt

  metrics      SoftArticleMetric[]

  @@index([companyId])
  @@index([status])
}

model SoftArticleMetric {
  id            String      @id @default(cuid())
  softArticleId String
  softArticle   SoftArticle @relation(fields: [softArticleId], references: [id], onDelete: Cascade)
  channel       String      // 渠道名称
  views         Int         @default(0)
  shares        Int         @default(0)
  comments      Int         @default(0)
  createdAt     DateTime    @default(now())
  updatedAt     DateTime    @updatedAt

  @@unique([softArticleId, channel])
}
```

### 2.5 活动/展会系统

```prisma
model Event {
  id            String   @id @default(cuid())
  companyId     String?
  company       Company? @relation(fields: [companyId], references: [id])
  title         String
  description   String?
  type          EventType
  startDate     DateTime
  endDate       DateTime
  location      String?  // 线下地点或线上链接
  coverImage    String?
  capacity      Int?     // 人数上限
  registrationDeadline DateTime?
  status        EventStatus @default(UPCOMING)
  isFeatured    Boolean  @default(false)
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  registrations EventRegistration[]
}

enum EventType {
  EXHIBITION     // 展会
  CONFERENCE     // 论坛/峰会
  WEBINAR        // 线上直播
  SALON          // 技术沙龙
  WORKSHOP       // 培训/工作坊
}

enum EventStatus {
  UPCOMING
  ONGOING
  COMPLETED
  CANCELLED
}

model EventRegistration {
  id        String   @id @default(cuid())
  eventId   String
  event     Event    @relation(fields: [eventId], references: [id], onDelete: Cascade)
  name      String
  email     String
  phone     String?
  company   String?
  position  String?
  createdAt DateTime @default(now())

  @@index([eventId])
}
```

### 2.6 硬核芯评选系统

```prisma
model Award {
  id            String    @id @default(cuid())
  title         String    // 评选活动标题
  description   String?   // 评选规则说明
  year          Int       // 评选年份
  startDate     DateTime
  endDate       DateTime
  status        AwardStatus @default(UPCOMING)
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt

  categories    AwardCategory[]
  nominations   AwardNomination[]
  votes         AwardVote[]
}

enum AwardStatus {
  UPCOMING      // 即将开始
  NOMINATING    // 提名中
  VOTING        // 投票中
  REVIEWING     // 评审中
  COMPLETED     // 已完成
}

model AwardCategory {
  id          String @id @default(cuid())
  awardId     String
  award       Award  @relation(fields: [awardId], references: [id], onDelete: Cascade)
  name        String // 奖项名称（如"年度最佳芯片"）
  description String?
  sort        Int    @default(0)

  @@unique([awardId, name])
}

model AwardNomination {
  id             String           @id @default(cuid())
  awardId        String
  award          Award            @relation(fields: [awardId], references: [id], onDelete: Cascade)
  categoryId     String
  category       AwardCategory    @relation(fields: [categoryId], references: [id])
  companyId      String
  company        Company          @relation(fields: [companyId], references: [id])
  productName    String?          // 参评产品
  description    String?          // 参评理由
  materials      Json?            // 参评材料附件
  status         NominationStatus @default(PENDING)
  createdAt      DateTime         @default(now())

  @@unique([awardId, categoryId, companyId])
}

enum NominationStatus {
  PENDING
  APPROVED
  REJECTED
}

model AwardVote {
  id        String   @id @default(cuid())
  awardId   String
  award     Award    @relation(fields: [awardId], references: [id], onDelete: Cascade)
  categoryId String?
  userId    String
  user      User     @relation(fields: [userId], references: [id])
  ip        String?
  createdAt DateTime @default(now())

  @@unique([awardId, categoryId, userId])
  @@index([awardId])
}
```

### 2.7 广告投放系统

```prisma
model AdPosition {
  id          String   @id @default(cuid())
  name        String   @unique  // 广告位名称（首页Banner/侧边栏/文章内嵌）
  code        String   @unique  // 标识码（home_banner/sidebar/article_embed）
  description String?
  width       Int
  height      Int
  price       Float?   // 参考价格
  maxCount    Int      @default(1)  // 同时投放数量上限
  status      Boolean  @default(true)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  ads         Ad[]
}

model AdCampaign {
  id             String         @id @default(cuid())
  companyId      String
  company        Company        @relation(fields: [companyId], references: [id])
  name           String
  budget         Float
  startDate      DateTime
  endDate        DateTime
  targetAudience Json?          // 目标受众标签
  status         CampaignStatus @default(DRAFT)
  createdAt      DateTime       @default(now())
  updatedAt      DateTime       @updatedAt

  ads            Ad[]
}

enum CampaignStatus {
  DRAFT
  ACTIVE
  PAUSED
  ENDED
}

model Ad {
  id             String      @id @default(cuid())
  positionId     String
  position       AdPosition  @relation(fields: [positionId], references: [id])
  campaignId     String?
  campaign       AdCampaign? @relation(fields: [campaignId], references: [id])
  title          String
  image          String?
  videoUrl       String?
  link           String
  companyId      String
  company        Company     @relation(fields: [companyId], references: [id])
  startDate      DateTime
  endDate        DateTime
  budget         Float?
  cost           Float?
  status         Status      @default(PENDING)
  impressions    Int         @default(0)
  clicks         Int         @default(0)
  conversionRate Float?      @default(0)
  abTestGroup    String?     // A/B 测试分组
  sort           Int         @default(0)
  createdAt      DateTime    @default(now())
  updatedAt      DateTime    @updatedAt

  @@index([positionId, status])
  @@index([companyId])
}
```

### 2.9 招聘系统

```prisma
model Job {
  id          String      @id @default(cuid())

  company     Company     @relation(fields: [companyId], references: [id])

  city        String?

enum JobType {
  FULL_TIME    // 全职
  PART_TIME    // 兼职

enum JobStatus {
  DRAFT
  PUBLISHED
  CLOSED
}

model JobApplication {
  id          String   @id @default(cuid())

  job         Job      @relation(fields: [jobId], references: [id], onDelete: Cascade)
  name        String
  email       String
  phone       String?
  resumeUrl   String?
  coverLetter String?
  status      ApplicationStatus @default(PENDING)

enum ApplicationStatus {
  PENDING
  REVIEWED
  INTERVIEWING
  ACCEPTED
  REJECTED
```

### 2.10 用户系统

```prisma
model User {
  id        String   @id @default(cuid())
  email     String   @unique
  password  String?  // 为空表示第三方登录
  name      String?
  avatar    String?
  phone     String?
  role      UserRole @default(USER)
  companyId String?  // 企业用户关联的企业
  company   Company? @relation(fields: [companyId], references: [id])
  status    UserStatus @default(ACTIVE)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  sessions  Session[]
  votes     AwardVote[]
  jobApplications JobApplication[]
}

enum UserRole {
  SUPER_ADMIN     // 超级管理员
  ADMIN           // 运营编辑
  ENTERPRISE      // 企业用户
  USER            // 普通注册用户
}

enum UserStatus {
  ACTIVE
  DISABLED
  BANNED
}

model Session {
  id        String   @id @default(cuid())
  userId    String
  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  token     String   @unique
  expiresAt DateTime
  createdAt DateTime @default(now())

  @@index([userId])
  @@index([token])
}
```

```prisma
model MediaChannel {

  size      Int?       // 文件大小（字节）

  result      String?  // 合作成果

  createdAt DateTime   @default(now())

  AD_IMPRESSIONS
```
